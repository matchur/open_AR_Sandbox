    """Colored log formatter."""

    def __init__(self, *args, colors: Optional[Dict[str, str]] = None, **kwargs) -> None:
        """Initialize the formatter with specified format strings."""

        super().__init__(*args, **kwargs)

        self.colors = colors if colors else {}


[documentos]    def format(self, record) -> str:
        """Format the specified record as text."""

        record.color = self.colors.get(record.levelname, '')
        record.reset = Style.RESET_ALL

        return super().format(record)

[documentos]def set_logger(name, level = logging.DEBUG):
    """
    Create a new handle in any destination
    Args:
        name: name of new handle
    Returns:
    """
    if name[:8] != "sandbox.":
        name = "sandbox."+name
    if not logging.getLogger(name).hasHandlers():
        logging.getLogger(name).addHandler(name)

    log = logging.getLogger(name)
    log.setLevel(level)
    return log
